💡 Las credenciales se guardan cifradas en localStorage.
Ejecuta este SQL en el editor de Supabase para crear las tablas: 
-- Capitanes
CREATE TABLE capitanes (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  nombre text NOT NULL,
  telefono text,
  created_at timestamptz DEFAULT now()
);

-- Lugares de encuentro
CREATE TABLE lugares (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  nombre text NOT NULL,
  direccion text,
  created_at timestamptz DEFAULT now()
);

-- Días del plan
CREATE TABLE dias_plan (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  fecha date NOT NULL UNIQUE,
  lugar_id uuid REFERENCES lugares(id) ON DELETE SET NULL,
  hora_manana time,
  capitan_manana_id uuid REFERENCES capitanes(id) ON DELETE SET NULL,
  hora_tarde time,
  capitan_tarde_id uuid REFERENCES capitanes(id) ON DELETE SET NULL,
  tiene_predicacion boolean DEFAULT true,
  created_at timestamptz DEFAULT now()
);

-- RLS: habilitar y crear políticas según tus necesidades
ALTER TABLE capitanes ENABLE ROW LEVEL SECURITY;
ALTER TABLE lugares ENABLE ROW LEVEL SECURITY;
ALTER TABLE dias_plan ENABLE ROW LEVEL SECURITY;

-- Política pública de lectura/escritura (ajusta según autenticación)
CREATE POLICY "allow_all" ON capitanes FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "allow_all" ON lugares FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "allow_all" ON dias_plan FOR ALL USING (true) WITH CHECK (true);
